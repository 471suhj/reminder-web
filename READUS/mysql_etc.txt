	identified by 'password'
	password expire never
	account unlock
	password history 0
	password reuse interval 0 day
	password require current;

GRANT SELECT, INSERT, UPDATE, DELETE, CREATE TEMPORARY TABLES, SHOW VIEW ON reminder_web.* TO 'user'@'localhost';
